Filtering

Filter Slides 

$('.filtering').slick({
    thumbs: true,
    slidesToShow: 2,
    slidesToScroll: 2
});
var filtered = false;
$('.js-filter').on('click', function() {
    if (filtered === false) {
        $('.filtering').slickFilter(':even');
        $(this).text('Unfilter Slides');
        filtered = true;
    } else {
        $('.filtering').slickUnfilter();
        $(this).text('Filter Slides');
        filtered = false;
    }
});
